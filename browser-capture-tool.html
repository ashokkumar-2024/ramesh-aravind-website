🎯 Canva Image Capture Tool

Capture images from each panel/section of the Canva site

📋 Instructions:

  1. Open the Canva site: https://ergherh.my.canva.site/dr-ra-website
  2. Open browser console (F12 or Cmd+Option+I)
  3. Copy and paste the script below into the console
  4. Press Enter to run it
  5. A floating button will appear on the page
  6. Scroll to each panel/section and click "Capture Panel" button
  7. After capturing all panels, click "Download JSON"
  8. Send me the downloaded JSON file

📜 Copy this script:

(function() { let panels = []; let currentPanel = 1; // Remove any existing tool const existing = document.getElementById('captureToolContainer'); if (existing) existing.remove(); // Create floating UI container const container = document.createElement('div'); container.id = 'captureToolContainer'; container.style.cssText = 'position: fixed; top: 20px; right: 20px; z-index: 999999; background: white; padding: 20px; border-radius: 10px; box-shadow: 0 4px 20px rgba(0,0,0,0.3); font-family: Arial; min-width: 300px;'; // Create elements const title = document.createElement('h3'); title.style.cssText = 'margin: 0 0 15px 0; color: #333;'; title.textContent = 'Image Capture Tool'; const panelInfo = document.createElement('div'); panelInfo.style.cssText = 'margin-bottom: 10px;'; const panelNumSpan = document.createElement('span'); panelNumSpan.id = 'panelNum'; panelNumSpan.textContent = '1'; panelInfo.innerHTML = 'Current Panel: '; panelInfo.appendChild(panelNumSpan); const panelNameInput = document.createElement('input'); panelNameInput.type = 'text'; panelNameInput.id = 'panelName'; panelNameInput.placeholder = 'Panel name (e.g., hero, about)'; panelNameInput.style.cssText = 'width: 100%; padding: 8px; margin-bottom: 10px; border: 1px solid #ddd; border-radius: 4px; box-sizing: border-box;'; const captureBtn = document.createElement('button'); captureBtn.id = 'captureBtn'; captureBtn.textContent = '📸 Capture This Panel'; captureBtn.style.cssText = 'width: 100%; padding: 12px; background: #4CAF50; color: white; border: none; border-radius: 5px; cursor: pointer; font-size: 14px; margin-bottom: 10px;'; const downloadBtn = document.createElement('button'); downloadBtn.id = 'downloadBtn'; downloadBtn.textContent = '💾 Download JSON'; downloadBtn.disabled = true; downloadBtn.style.cssText = 'width: 100%; padding: 12px; background: #2196F3; color: white; border: none; border-radius: 5px; cursor: pointer; font-size: 14px; margin-bottom: 10px; opacity: 0.5;'; const statusDiv = document.createElement('div'); statusDiv.id = 'statusMsg'; statusDiv.style.cssText = 'margin-top: 10px; padding: 10px; border-radius: 5px; font-size: 12px; display: none;'; const capturedDiv = document.createElement('div'); capturedDiv.id = 'capturedList'; capturedDiv.style.cssText = 'margin-top: 10px; font-size: 12px; color: #666;'; capturedDiv.textContent = 'No panels captured yet'; // Append all elements container.appendChild(title); container.appendChild(panelInfo); container.appendChild(panelNameInput); container.appendChild(captureBtn); container.appendChild(downloadBtn); container.appendChild(statusDiv); container.appendChild(capturedDiv); document.body.appendChild(container); // Functions function showStatus(message, isError = false) { statusDiv.textContent = message; statusDiv.style.background = isError ? '#f8d7da' : '#d4edda'; statusDiv.style.color = isError ? '#721c24' : '#155724'; statusDiv.style.display = 'block'; setTimeout(() => { statusDiv.style.display = 'none'; }, 3000); } function updateCapturedList() { if (panels.length === 0) { capturedDiv.textContent = 'No panels captured yet'; } else { capturedDiv.innerHTML = 'Captured:
' + panels.map(p => `${p.panel}: ${p.name} (${p.images.length} images)`).join('
'); } } // Event handlers captureBtn.onclick = function() { const panelName = panelNameInput.value.trim() || `panel-${currentPanel}`; const images = Array.from(document.querySelectorAll('img')) .filter(img => img.src && img.src.includes('_assets/media/')) .map(img => { const rect = img.getBoundingClientRect(); return { src: img.src, filename: img.src.split('/').pop(), visible: rect.top >= 0 && rect.bottom <= window.innerHeight && rect.width > 50 && rect.height > 50, top: Math.round(rect.top + window.scrollY), left: Math.round(rect.left), width: Math.round(rect.width), height: Math.round(rect.height) }; }); const visibleImages = images.filter(img => img.visible); if (visibleImages.length === 0) { showStatus('⚠️ No images found in viewport! Make sure images are fully visible.', true); return; } panels.push({ panel: currentPanel, name: panelName, imageCount: visibleImages.length, images: visibleImages.map(img => img.filename), allImageData: visibleImages }); showStatus(`✅ Captured ${visibleImages.length} images from "${panelName}"`); currentPanel++; panelNumSpan.textContent = currentPanel; panelNameInput.value = ''; downloadBtn.disabled = false; downloadBtn.style.opacity = '1'; updateCapturedList(); }; downloadBtn.onclick = function() { const data = { totalPanels: panels.length, totalImages: panels.reduce((sum, p) => sum + p.imageCount, 0), capturedAt: new Date().toISOString(), panels: panels }; const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'canva-panels-mapping.json'; a.click(); URL.revokeObjectURL(url); showStatus('✅ Downloaded JSON file!'); }; showStatus('🎯 Ready! Scroll to first panel and click Capture'); console.log('✅ Capture tool loaded! Look for the floating panel on the right.'); })();

💡 Tips:

  • Make sure you can see the images in the panel before clicking "Capture"
  • Give each panel a descriptive name (hero, about, awards, etc.)
  • The tool only captures images visible in your viewport
  • After capturing all panels, download the JSON and send it to me
